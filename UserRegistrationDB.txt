CREATE PROCEDURE [dbo].[spUserRegistration]

@LastName			nvarchar(100),

@PhoneNumber		nvarchar(15),
@CreditCardNumber	nvarchar(max),
@CVV				nvarchar(max),
@ExpireDate			nvarchar(50)

BEGIN

	)
